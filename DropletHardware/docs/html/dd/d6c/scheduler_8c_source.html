77 // Adds a new task to the task queue
78 // time is number of milliseconds until function is executed
79 // function is a function pointer to execute
80 // arg is the argument to supply to function
81 Task_t* schedule_task(volatile uint32_t time, void (*function)(void*), void* arg)
82 {
83  Task_t* new_task = (Task_t*)malloc(sizeof(Task_t));
84  if (new_task == NULL) return NULL;
85 
86  new_task->scheduled_time = time + get_time();
87  if ((uint16_t)(new_task->scheduled_time) < 2) new_task->scheduled_time += 4;
88  new_task->arg = arg;
89  new_task->task_function = function;
90 
91  // Turn off interrupts so we don't muck up the task list during this function
92  ATOMIC_BLOCK(ATOMIC_RESTORESTATE)
93  {
94  // Find the new task's proper spot in the list of tasks
95  // task_list is a linked list sorted by scheduled_time, smallest first
96  new_task->next = task_list;
97 
98  // If the new task is the next to be executed, put it at the front of the list
99  if (task_list == NULL || new_task->scheduled_time <= task_list->scheduled_time)
100 
101  {
102  task_list = new_task;
103  // If scheduled_time is in the current epoch, set the RTC compare interrupt
104  if (new_task->scheduled_time <= ((((uint32_t)rtc_epoch) << 16) | (uint32_t)RTC.PER))
105  {
106  while (RTC.STATUS & RTC_SYNCBUSY_bm);
107  RTC.COMP = (uint16_t)(new_task->scheduled_time);
108  RTC.INTCTRL |= RTC_COMPINTLVL_LO_gc;
109  }
110  else
111  {
112  RTC.INTCTRL &= ~RTC_COMPINTLVL_LO_gc;
113  }
114  }
115  // If the new task is not the next to be executed, iterate through the task_list,
116  // find its position in the linked list, and insert it there.
117  else
118  {
119  Task_t* tmp_task_ptr = task_list;
120  while (tmp_task_ptr->next != NULL && new_task->scheduled_time > tmp_task_ptr->next->scheduled_time)
121  {
122  tmp_task_ptr = tmp_task_ptr->next;
123  }
124  new_task->next = tmp_task_ptr->next;
125  tmp_task_ptr->next = new_task;
126  }
127 
128 
129  num_tasks++;
130  }
131 
132  return new_task;
133 }

174 {
175  ATOMIC_BLOCK(ATOMIC_RESTORESTATE) // Disable interrupts
176  {
177  // Run all tasks that are scheduled to execute in the next 2ms
178  // (The RTC compare register takes 2 RTC clock cycles to update)
179  while (task_list != NULL && task_list->scheduled_time <= get_time() + 2)
180  {
181  if (SCHEDULER_DEBUG_MODE >= 1) printf("Executing task %p (%p)\r\n", task_list, task_list->task_function);
182  Task_t* cur_task = task_list;
183  task_list = cur_task->next;
184  NONATOMIC_BLOCK(NONATOMIC_FORCEOFF) // Enable interrupts during tasks
185  {
186  cur_task->task_function(cur_task->arg); // run the task
187  }
188  free(cur_task);
189  cur_task = NULL;
190  num_tasks--;
191  }
192  // If the next task to be executed is in the current epoch, set the RTC compare register and interrupt
193  if (task_list != NULL && task_list->scheduled_time <= ((((uint32_t)rtc_epoch) << 16) | (uint32_t)RTC.PER))
194  {
195  while (RTC.STATUS & RTC_SYNCBUSY_bm);
